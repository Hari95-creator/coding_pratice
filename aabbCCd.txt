String s = "aabbccd";
int count = 0;
char currentChar = s.charAt(0);

String compressedString = "";

for (int i = 0; i < s.length(); i++) {
    if (s.charAt(i) == currentChar) {
        count++;
    } else {
        compressedString += currentChar + String.valueOf(count);
        currentChar = s.charAt(i);
        count = 1;
    }
}

// Append the last group
compressedString += currentChar + String.valueOf(count);

System.out.println(compressedString);  // Output: a2b2c2d1
